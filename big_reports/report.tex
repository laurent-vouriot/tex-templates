	\subsection{Algorithm}

	\begin{algorithm}[H]
		\caption{Kruskal(G)}
		\begin{algorithmic}[1]
			\State{$A \gets \emptyset $}
			\ForAll{verticies $v\in G$}
				\State{$Set[] \gets v$}
			\EndFor
			\State{\textbf{Sort} $Set$}
			\ForAll{edge $(u, v) \in G$}
				\If {$find(u) \neq find(v)$}
					\State{$A \Union (u,v)$}
				\EndIf
			\EndFor\\
			\Return $A$
		\end{algorithmic}
	\end{algorithm}
